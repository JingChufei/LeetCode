class Solution:
    def searchRange(self, nums: List[int], target: int) -> List[int]:

        if not nums:
            return [-1, -1]

        def find_start(nums, target):
            l = 0
            r = len(nums) - 1

            while l < r:
                mid = (l + r) // 2
                if nums[mid] == target:
                    r = mid
                elif nums[mid] > target:
                    r = mid - 1
                else:
                    l = mid + 1
            if nums[l] != target:
                return -1
            else:
                return l

        def find_end(nums, target):
            l = 0
            r = len(nums) - 1

            while l < r:
                mid = (l + r + 1) // 2
                if nums[mid] == target:
                    l = mid
                elif nums[mid] > target:
                    r = mid - 1
                else:
                    l = mid + 1
            if nums[l] != target:
                return -1
            else:
                return l

        start = find_start(nums, target)
        if start == -1:
            return [-1, -1]
        else:
            end = find_end(nums, target)
            return [start, end]
