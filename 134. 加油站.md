```python3
class Solution:
    def canCompleteCircuit(self, gas: List[int], cost: List[int]) -> int:
        
        n = len(gas)
        
        start = 0
        
        total = 0
        current = 0
        
        for i in range(n):
            
            total += gas[i] - cost[i]
            current += gas[i] - cost[i]
            
            if current < 0:
                start = i + 1
                current = 0
                
        if total >= 0:
            return start
        
        else:
            return -1
```
